## Overview

Three Jupyter notebooks that transform the Phase 2 catalog (118 profiles, 341 relationships, table annotations) into actionable analytical context. The goal is to produce structured summary reports (JSON + markdown) that a downstream EDA agent or analyst can consume — answering "can we trust this data?" and "what business questions can it support?"

**Scope**: Tier 1 tables only (11 tables, ~370K total rows). Techniques from Data 100 (pandas EDA, summary stats, missing data analysis, visualization) and Data 101 (referential integrity, granularity, outlier detection, entity resolution). No ML, no modeling, no clustering.

**Output philosophy**: Every notebook produces both human-readable markdown reports AND machine-readable JSON. The JSON becomes context for the future EDA agent. The markdown becomes reference documentation.

#### 1. Common preamble helper: `scripts/eda_helpers.py`

**File**: `scripts/eda_helpers.py` (new)

Shared utility functions imported by all three notebooks. Keeps notebooks focused on analysis, not boilerplate.

```python
"""Shared helpers for EDA notebooks."""
import json
import pandas as pd
from pathlib import Path
from config import TABLES_DIR, CATALOG_DIR, PROFILES_DIR

# --- Data Loading ---

def load_table(table_name: str) -> pd.DataFrame:
    """Load a CSV table with consistent NA handling."""
    path = TABLES_DIR / f"{table_name}.csv"
    return pd.read_csv(path, keep_default_na=False, na_values=[""], low_memory=False)

def load_profile(table_name: str) -> dict:
    """Load a table's JSON profile from catalog."""
    with open(PROFILES_DIR / f"{table_name}.json") as f:
        return json.load(f)

def load_all_profiles() -> dict:
    """Load all profiles keyed by table name."""
    profiles = {}
    for f in sorted(PROFILES_DIR.glob("*.json")):
        with open(f) as fh:
            p = json.load(fh)
            profiles[p["table_name"]] = p
    return profiles

def load_relationships() -> list:
    """Load relationships from catalog."""
    with open(CATALOG_DIR / "relationships.json") as f:
        return json.load(f)

def load_annotations() -> dict:
    """Load table annotations from catalog."""
    with open(CATALOG_DIR / "table_annotations.json") as f:
        return json.load(f)

# --- Tier Filtering ---

TIER_1_TABLES = [
    "tblProject", "tblTimesheet", "tblContact", "tblCompany",
    "tblBid", "tblBill", "tblBillHistory", "tblBillHistDetail",
    "tblPO", "tblPurchasing", "tblSheetNEW",
]

# --- Output Helpers ---

def save_eda_output(data: dict, name: str, eda_dir: Path):
    """Save EDA results as both JSON and markdown."""
    eda_dir.mkdir(parents=True, exist_ok=True)
    # JSON
    with open(eda_dir / f"{name}.json", "w") as f:
        json.dump(data, f, indent=2, default=str)
    # Markdown (caller provides md content in data["_markdown"])
    if "_markdown" in data:
        with open(eda_dir / f"{name}.md", "w") as f:
            f.write(data["_markdown"])
```

**Cell 2 — Section: Referential Integrity**:

For each Tier 1 table, check all high-confidence FK relationships:
- Load the FK column values from the source table
- Load the PK column values from the target table
- Compute: orphan count, orphan rate, sample orphan values

```python
# Filter to high-confidence relationships involving Tier 1 tables
tier1_set = set(TIER_1_TABLES)
fk_checks = []

for rel in relationships:
    if rel["confidence"] != "high":
        continue
    # Check both directions — we want cases where a Tier 1 table references another table
    if rel["from_table"] in tier1_set:
        fk_checks.append({
            "source_table": rel["from_table"],
            "source_column": rel["from_column"],
            "target_table": rel["to_table"],
            "target_column": rel["to_column"],
            "join_type": rel.get("join_type", "unknown"),
        })

print(f"FK relationships to check: {len(fk_checks)}")

# Run integrity checks
integrity_results = []
for check in fk_checks:
    src_tbl = check["source_table"]
    tgt_tbl = check["target_table"]
    src_col = check["source_column"]
    tgt_col = check["target_column"]

    # Load source — already in memory if Tier 1
    if src_tbl in tables:
        src_df = tables[src_tbl]
    else:
        try:
            src_df = load_table(src_tbl)
        except FileNotFoundError:
            continue

    # Load target
    if tgt_tbl in tables:
        tgt_df = tables[tgt_tbl]
    else:
        try:
            tgt_df = load_table(tgt_tbl)
        except FileNotFoundError:
            continue

    if src_col not in src_df.columns or tgt_col not in tgt_df.columns:
        continue

    # Get non-null FK values from source
    fk_values = src_df[src_col].dropna()
    if len(fk_values) == 0:
        continue

    # Get valid PK values from target
    pk_values = set(tgt_df[tgt_col].dropna().astype(str))
    fk_values_str = fk_values.astype(str)

    orphans = fk_values_str[~fk_values_str.isin(pk_values)]
    orphan_rate = len(orphans) / len(fk_values) * 100

    result = {
        "source_table": src_tbl,
        "source_column": src_col,
        "target_table": tgt_tbl,
        "target_column": tgt_col,
        "total_fk_values": len(fk_values),
        "orphan_count": len(orphans),
        "orphan_rate_pct": round(orphan_rate, 2),
        "sample_orphans": orphans.head(5).tolist() if len(orphans) > 0 else [],
    }
    integrity_results.append(result)

# Display worst offenders
integrity_df = pd.DataFrame(integrity_results)
print(f"\n=== FK Integrity Check Results ({len(integrity_results)} relationships) ===")
if len(integrity_df) > 0:
    worst = integrity_df[integrity_df["orphan_count"] > 0].sort_values("orphan_rate_pct", ascending=False)
    print(f"Relationships with orphans: {len(worst)}")
    if len(worst) > 0:
        print(worst[["source_table", "source_column", "target_table", "orphan_count", "orphan_rate_pct"]].head(20).to_string(index=False))
```

**Cell 3 — Section: Null Pattern Analysis**:

For each Tier 1 table, go beyond simple null counts. Analyze whether nulls are **systematic** (entire columns null for certain record types) or **random** (scattered).

```python
null_analysis = {}

for name in TIER_1_TABLES:
    df = tables[name]
    profile = profiles[name]
    cols = profile["columns"]

    # Classify columns by null severity
    high_null = []    # >50% null
    moderate_null = []  # 10-50% null
    low_null = []     # 1-10% null
    complete = []     # <1% null

    for col in cols:
        pct = col["null_pct"]
        entry = {"column": col["name"], "null_pct": pct, "sql_type": col["sql_type"]}
        if pct > 50:
            high_null.append(entry)
        elif pct > 10:
            moderate_null.append(entry)
        elif pct > 1:
            low_null.append(entry)
        else:
            complete.append(entry)

    null_analysis[name] = {
        "total_columns": len(cols),
        "high_null_columns": high_null,
        "moderate_null_columns": moderate_null,
        "low_null_columns": low_null,
        "complete_columns": complete,
        "high_null_count": len(high_null),
        "moderate_null_count": len(moderate_null),
        "completeness_score": round(len(complete) / len(cols) * 100, 1),
    }
    print(f"{name}: {len(complete)}/{len(cols)} complete columns ({null_analysis[name]['completeness_score']}%), "
          f"{len(high_null)} high-null, {len(moderate_null)} moderate-null")
```

**Cell 4 — Section: Temporal Coverage**:

For each Tier 1 table, find all date columns, compute their min/max range, and identify the **usable data window**.

```python
temporal_coverage = {}
date_types = {"smalldatetime", "datetime", "datetime2", "date"}

for name in TIER_1_TABLES:
    df = tables[name]
    profile = profiles[name]
    date_cols = [c for c in profile["columns"] if c["sql_type"] in date_types]

    if not date_cols:
        temporal_coverage[name] = {"date_columns": [], "usable_window": None}
        continue

    col_ranges = []
    for col_info in date_cols:
        col_name = col_info["name"]
        if col_name not in df.columns:
            continue
        series = pd.to_datetime(df[col_name], errors="coerce")
        non_null = series.dropna()
        if len(non_null) == 0:
            continue
        col_ranges.append({
            "column": col_name,
            "min_date": str(non_null.min().date()),
            "max_date": str(non_null.max().date()),
            "non_null_count": len(non_null),
            "null_pct": round((1 - len(non_null) / len(df)) * 100, 1),
        })

    # Overall usable window = intersection of primary date columns
    all_mins = [pd.Timestamp(c["min_date"]) for c in col_ranges if c["non_null_count"] > len(df) * 0.5]
    all_maxs = [pd.Timestamp(c["max_date"]) for c in col_ranges if c["non_null_count"] > len(df) * 0.5]

    temporal_coverage[name] = {
        "date_columns": col_ranges,
        "overall_min": str(min(all_mins).date()) if all_mins else None,
        "overall_max": str(max(all_maxs).date()) if all_maxs else None,
    }
    if col_ranges:
        print(f"{name}: {len(col_ranges)} date cols, range {col_ranges[0]['min_date']} to {col_ranges[0]['max_date']}")
```

**Cell 5 — Section: Duplicate Detection**:

Check for duplicate rows on primary key columns and on likely-unique business columns.

```python
duplicate_results = {}

for name in TIER_1_TABLES:
    df = tables[name]
    profile = profiles[name]

    # Find PK column
    pk_cols = [c["name"] for c in profile["columns"] if c.get("is_primary_key")]

    dupes = {}
    if pk_cols:
        pk_col = pk_cols[0]
        pk_dupes = df[pk_col].duplicated().sum()
        dupes["pk_column"] = pk_col
        dupes["pk_duplicates"] = int(pk_dupes)

    # Full row duplicates
    full_dupes = df.duplicated().sum()
    dupes["full_row_duplicates"] = int(full_dupes)

    duplicate_results[name] = dupes
    if pk_dupes > 0 or full_dupes > 0:
        print(f"WARNING: {name} has {pk_dupes} PK dupes, {full_dupes} full-row dupes")

print("\nDuplicate check complete.")
```

```python
# Goal: compare tblBid.TotProjCost to tblProject.TotProjCost for awarded bids

bid["TotProjCost"] = pd.to_numeric(bid["TotProjCost"], errors="coerce")
bid["JobNo"] = pd.to_numeric(bid["JobNo"], errors="coerce")

    awarded["variance_pct"] = (awarded["variance"] / awarded["TotProjCost"] * 100).round(1)
```
